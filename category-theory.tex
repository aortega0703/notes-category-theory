\section{Category}

For a category $C$
\begin{itemize}
  \item Object:\\
    $ob_C \coloneqq$ collection of objects in $C$ 
  \item Morphism:\\
    Arrow between objects\\
    $hom_C(a, b) \coloneqq$ set of morphisms from $a$ to $b$, $\forall a, b \in ob_C$\\
    $f: a \to b$ in $C$ means $f \in hom_C(a,c)$ 
  \item Identity:\\
    $id_c: c \to c,\ \forall c \in ob_C$
  \item Composition:\\
    $g \circ f : a \to c,\ \forall a, b, c \in ob_C, f: a \to b,\ g: b \to c$
\end{itemize}

s.t. satisfies
\begin{itemize}
  \item Unitality:\\
    $id_b \circ f = f \circ id_a = f,\ \forall a, b \in ob_C, f: a \to b$
  \item Assosiativity:\\
    $h \circ (g \circ f) = (h \circ g) \circ f,\ \forall a,b,c\in ob_C, f:a\to b,\ g:b\to c,\ h:c\to d$
\end{itemize}

e.g.
\[\begin{tikzcd}
  a \arrow[out=90, in=180, loop, swap, "id_a", looseness=5]
    \arrow[r, "f"]
    \arrow[dr, swap, "g \circ f"]
  &
  b \arrow[out=0, in=90, loop, swap, "id_b", looseness=4.5]
    \arrow[d, "g"]\\
  &
  c \arrow[out=-90, in=0, swap, "id_c", looseness=5]
\end{tikzcd}\]

\section{Functor}

For a functor $F: C \to D$ with $C$ and $D$ categories
\[\begin{tikzcd}
  C \arrow[r, "F"]
  &
  D
\end{tikzcd}\]
%
\begin{itemize}
  \item Object to objects\\
    $F(c) \in ob_D,\ \forall c \in ob_C$
  \item Morphism to morphisms\\
    $F(f): F(c) \to F(c'),\ \forall c,c' \in ob_C, f : c \to c'$
\end{itemize}

s.t. preserves
\begin{itemize}
  \item Identity\\
    $F(id_c) = id_{F(c)},\ \forall c \in ob_C$
  \item Composition\\
    $F(g \circ f) = F(g) \circ F(f),\ \forall a,b,c \in ob_C, f: a \to b,\ g: b \to c$
\end{itemize}

\section{Product category}
For a product category $C\times D$ with categories $C,\ D$
% https://q.uiver.app/?q=WzAsMyxbMSwwLCJDIFxcdGltZXMgRCJdLFswLDEsIkMiXSxbMiwxLCJEIl0sWzAsMSwiZnN0IiwyXSxbMCwyLCJzY24iXV0=
\[\begin{tikzcd}
	& {C \times D} \\
	C && D
	\arrow["fst"', from=1-2, to=2-1]
	\arrow["scn", from=1-2, to=2-3]
\end{tikzcd}\]
\begin{itemize}
  \item Object:\\
    $(c, d) \in ob_{C\times D},\ \forall c \in ob_C, d \in ob_D$
  \item Morphism:\\
    $(f,g) : (c,d) \to (c', d'),\ \forall f: c \to c',\ g:d \to d'$ 
\end{itemize}
