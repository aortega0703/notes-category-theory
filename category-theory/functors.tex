\section{Functors}

\subsection{Definition}
For a functor $F: C \to D$ with $C$ and $D$ categories:
\parencite{fong:7sketches}
\[\begin{tikzcd}
  C \arrow[r, "F"]
  &
  D
\end{tikzcd}\]
%
\begin{itemize}
  \item Object to objects\\
    $F(c) \in \mathrm{ob}_D,\ \forall(c \in \mathrm{ob}_C)$
  \item Morphism to morphisms\\
    $\mathrm{hom}_D \ni F(f): F(c) \to F(c'),\ \forall(c,c' \in \mathrm{ob}_C, f : c \to c')$
\end{itemize}

s.t. preserves
\begin{itemize}
  \item Identity\\
    $F(\mathrm{id}_c) = \mathrm{id}_{F(c)},\ \forall(c \in \mathrm{ob}_C)$
  \item Composition\\
    $F(g \circ f) = F(g) \circ F(f),
    \ \forall(a,b,c \in \mathrm{ob}_C, f: a \to b,\ g: b \to c)$
\end{itemize}

A functor $F:C\to D$ is said to be:
\begin{itemize}
  \item Faithful: \parencite{awodey:category_theory}\\
    If for every $x,y\in \mathrm{ob}_C$, the function $F_{x, y}: \mathrm{hom}_C(x, y) \to \mathrm{hom}_D(F(x), F(y))$
    is injective.
  \item Full: \parencite{adamek_herrlich_strecker:joy_cats}\\
    If for every $x,y\in \mathrm{ob}_C$, the function $F_{x, y}: \mathrm{hom}_C(x, y) \to \mathrm{hom}_D(F(x), F(y))$
    is surjective.  \item Endofunctor: \parencite{adamek_herrlich_strecker:joy_cats}\\
    If $D = C$, i.e, it is a functor from a category into itself.
  \item Bifunctor: \parencite{maclane:working_mathematician}\\
    If $C = A\times B$ for some categories $A$ and $B$, i.e. a fuctor of two
    variables.
\end{itemize}

\subsection{Important Functors}
\begin{itemize}
  \item Diagonal functor: \parencite{awodey:category_theory}\\
    $\Delta: C\to C\times C$
    \begin{itemize}
      \item Object/Morphisms:\\
        Maps every object/morphism $c$ in $C$ into $(c, c)$
    \end{itemize}

  \item Constant functor: \parencite{awodey:category_theory}\\
    $\Delta d: C\to D$
    \begin{itemize}
      \item Objects:\\
        Maps every object in $C$ to a fixed object $d\in \mathrm{ob}_D$.
      \item Morphisms:\\
        Maps every morphism in $C$ to $\mathrm{id}_d$.
    \end{itemize}

  \item Hom functor: \parencite{maclane:working_mathematician}\\
    $\mathrm{Hom}_C : C^{op}\times C \to \mathrm{Set}$
    \begin{itemize}
      \item Objects:\\
        From a pair of objects to a set of morphisms\\
        $\mathrm{Hom}_C (c', d) = \mathrm{hom}_C (c', d),\ \forall (c', d\in \mathrm{ob}_C)$
      \item Morphisms:\\
        From a morphism to a function of morphisms.\\
        $\mathrm{Hom}_C (f) : \mathrm{hom}_{C^{op}\times C}(c', d) \to \mathrm{hom}_{C^{op}\times C}(c, d'),
        \ \forall (f=(g^{op}, h): (c', d)\to (c, d'))$\\
