\section{Category}

\cite{7sketches_compositionality} For a category $C$:
\begin{itemize}
  \item Object:\\
    $ob_C \coloneqq$ collection of objects in $C$
  \item Morphism:\\
    Arrow between objects\\
    $hom_C(a, b) \coloneqq$ set of morphisms from $a$ to $b$,
    $\forall a, b \in ob_C$\\
    $f: a \to b$ in $C$ means $f \in hom_C(a,c)$
  \item Identity:\\
    $\mathrm{id}_c: c \to c,\ \forall c \in ob_C$
  \item Composition:\\
    $g \circ f : a \to c,\ \forall a, b, c \in ob_C, f: a \to b,\ g: b \to c$
\end{itemize}

s.t. satisfies
\begin{itemize}
  \item Unitality:\\
    $\mathrm{id}_b \circ f = f \circ \mathrm{id}_a = f,\ \forall a, b \in ob_C, f: a \to b$
  \item Assosiativity:\\
    $h \circ (g \circ f) = (h \circ g) \circ f,\
    \forall a,b,c\in ob_C, f:a\to b,\ g:b\to c,\ h:c\to d$
\end{itemize}

e.g.
\[\begin{tikzcd}
  a \arrow[out=90, in=180, loop, swap, "\mathrm{id}_a", looseness=5]
    \arrow[r, "f"]
    \arrow[dr, swap, "g \circ f"]
  &
  b \arrow[out=0, in=90, loop, swap, "\mathrm{id}_b", looseness=4.5]
    \arrow[d, "g"]\\
  &
  c \arrow[out=-90, in=0, swap, "\mathrm{id}_c", looseness=5]
\end{tikzcd}\]

\section{Functor}

\cite{7sketches_compositionality}
For a functor $F: C \to D$ with $C$ and $D$ categories:
\[\begin{tikzcd}
  C \arrow[r, "F"]
  &
  D
\end{tikzcd}\]
%
\begin{itemize}
  \item Object to objects\\
    $F(c) \in ob_D,\ \forall c \in ob_C$
  \item Morphism to morphisms\\
    $F(f): F(c) \to F(c'),\ \forall c,c' \in ob_C, f : c \to c'$
\end{itemize}

s.t. preserves
\begin{itemize}
  \item Identity\\
    $F(\mathrm{id}_c) = \mathrm{id}_{F(c)},\ \forall c \in ob_C$
  \item Composition\\
    $F(g \circ f) = F(g) \circ F(f),
    \ \forall a,b,c \in ob_C, f: a \to b,\ g: b \to c$
\end{itemize}

\section{Cartesian Product}

\subsection{Between objects}
\cite{nlab:cartesian_product}
In a category $C$ the cartesian product of $2$ objects $a,\ b$ is another
object $a\times b$ with morphisms $\mathrm{fst}: a\times b \to a,
\ \mathrm{snd}: a\times b \to b$ such that, for any other object $c$ with the
same property there exist a unique $f:c \to a\times b$ such that the following
diagram commutes

\[\begin{tikzcd}
	& {a\times b} \\
	a && b \\
	& c
	\arrow["{\mathrm{fst}}"', from=1-2, to=2-1]
	\arrow["{\mathrm{snd}}", from=1-2, to=2-3]
	\arrow["f"', dashed, from=3-2, to=1-2]
	\arrow["g", from=3-2, to=2-1]
	\arrow["h"', from=3-2, to=2-3]

The product candidates for $2\times 4$ are $2$ with $\mathrm{id}_2 : 2\to 2$
and $d\circ c:2\to 4$, $3$ with $e: 3\to 2$ and $c\circ d\circ e: 3\to 4$,
and $6$ with $e \circ g: 6\to 2$ and $c\circ f$. The product $2\times 4$
corresponds then to the object $2$. The cartesian product in a
preorder corresponds to the meet operation $\land$ or infimum.

\subsection{Between categories}
\cite{nlab:product_category} For a product category $C\times D$ with categories
$C,\ D$:
\begin{itemize}
  \item Object:\\
    $(c, d) \in ob_{C\times D},\ \forall c \in ob_C, d \in ob_D$
  \item Morphism:\\
    $(f,g) : (c,d) \to (c', d'),\ \forall f: c \to c',\ g:d \to d'$
\end{itemize}
This operation is the cartesian product in the category of categories Cat

\section{Hom functor}
\cite{nlab:hom-functor}$Hom_C : C^{op}\times C \to Set$
\begin{itemize}

    From a pair of objects to a set of morphisms\\
    $Hom_C (c', d) = hom_C (c', d),\ \forall c', d\in ob_C$
  \item Morphism:\\
    From a morphism to a function of morphisms.\\
