\section{Natural Transformations}

\subsection{Definition}
A natural transformation $\alpha : F \Rightarrow G$ from a functor $F:C \to D$
to a functor $G: C \to D$ consists of: \parencite{fong_spivak:7sketches}

\begin{itemize}
  \item $c$-Components:\\
    $(\alpha_c : F(c) \to G(c)) \in D,\ (\forall c\in \mathrm{ob}_C)$\\
    If every $c$-component is an isomorphism the natural transformation is then
    called a natural isomorphism.
  \item Naturality Condition:\\
    The following diagram must commute\footnote{Parallel morphisms i.e. those
    with the same source and target, are equal} for all $c,c'\in C$ and $(f:c\to
    c')\in C$:

\subsection{Identity}
For any categories $C,D$ with a functor $F:C\to D$, there exists a natural
transformation $\mathrm{id}_F: F\Rightarrow F$ which consists of:
\parencite{leinster:basic_category_theory}
\begin{itemize}
  \item $c$-Components:\\
    $(\mathrm{id}_F)_c: F(c) \to F(c) = \mathrm{id}_{F(c)},\ (\forall c\in C)$
  \item Naturality Condition:
    % https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMpIl0sWzEsMCwiRihjJykiXSxbMCwxLCJGKGMpIl0sWzEsMSwiRihjJykiXSxbMCwxLCJGKGYpIl0sWzAsMiwiXFxtYXRocm17aWR9X3tGKGMpfSIsMl0sWzIsMywiRihmKSIsMl0sWzEsMywiXFxtYXRocm17aWR9X3tGKGMpfSJdXQ==
    \[\begin{tikzcd}
      {F(c)} & {F(c')} \\
      {F(c)} & {F(c')}
      \arrow["{F(f)}", from=1-1, to=1-2]
      \arrow["{\mathrm{id}_{F(c)}}"', from=1-1, to=2-1]
      \arrow["{F(f)}"', from=2-1, to=2-2]
      \arrow["{\mathrm{id}_{F(c)}}", from=1-2, to=2-2]
    \end{tikzcd}\]
    Which commutes for all $c,c'\in C$ and $(f:c\to c')\in C$ by using
    unitality of $D$.
\end{itemize}

\subsection{Composition}
\subsubsection*{Vertical Composition}
For categories $C,D$ with functors $F,G,H:C\to D$ and natural transformations
$\alpha:F\to G,\ \beta:G\to H$ there exists a natural transformation $\beta\circ
\alpha$ which consists of: \parencite{leinster:basic_category_theory}

\begin{itemize}
  \item $c$-Components:\\
    $(\beta\circ\alpha)_c: F(c) \to H(c) = \beta_c \circ \alpha_c,\ (\forall
    c\in C)$
  \item Naturality Condition:
    % https://q.uiver.app/?q=WzAsNixbMCwwLCJGKGMpIl0sWzEsMCwiRihjJykiXSxbMCwxLCJHKGMpIl0sWzAsMiwiSChjKSJdLFsxLDEsIkcoYycpIl0sWzEsMiwiSChjJykiXSxbMCwxLCJGKGYpIl0sWzIsNCwiRyhmKSJdLFszLDUsIkgoZikiXSxbMCwyLCJcXGFscGhhX2MiLDJdLFsyLDMsIlxcYmV0YV9jIiwyXSxbMSw0LCJcXGFscGhhX3tjJ30iXSxbNCw1LCJcXGJldGFfe2MnfSJdXQ==
    \[\begin{tikzcd}
      {F(c)} & {F(c')} \\
      {G(c)} & {G(c')} \\
      {H(c)} & {H(c')}
      \arrow["{F(f)}", from=1-1, to=1-2]
      \arrow["{G(f)}", from=2-1, to=2-2]
      \arrow["{H(f)}", from=3-1, to=3-2]
      \arrow["{\alpha_c}"', from=1-1, to=2-1]
      \arrow["{\beta_c}"', from=2-1, to=3-1]
      \arrow["{\alpha_{c'}}", from=1-2, to=2-2]
      \arrow["{\beta_{c'}}", from=2-2, to=3-2]
    \end{tikzcd}\]
    Which commutes for all $c,c'\in C$ and $(f:c\to c')\in C$ by using the
    naturality conditions of both $\alpha$ and $\beta$.
\end{itemize}
