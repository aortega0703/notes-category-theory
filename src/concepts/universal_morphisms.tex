\section{Universal Morphisms}
A universal morphism is either an initial or terminal morphism.
\parencite{maclane:working_mathematician}

\subsection{Initial morphisms}
For categories $C,D$, an initial morphism or a universal morphism from an object
$d\in D$ to a functor $F:C\to D$ consists of an object $c\in C$ coupled with a
morphism $f: d\to F(c)$ such that for any other $c'\in C$ and $f': d \to F(c')$,
there exists a unique $h:c\to c'$ such that the following diagram commutes.

\subsubsection*{Uniqueness up to isomorphism}
An initial morphism can be viewed as an initial object in the comma category
$(\Delta_d\downarrow F)$, and initial objects are unique up to isomorphism.
Therefore, initial morphisms are unique up to isomorphism.

\subsubsection*{As the initial object of a comma category}
Consider categories $C, D$ with a functor $F: C\to D$, together with the
discrete category $\underline{1}$ with a single element $\bullet$ and a constant
funtor $\Delta_d: \underline{1} \to D$. The comma category $(\Delta_d\downarrow
F)$ consists of:
\begin{itemize}
	\item Objects:\\
		$(\bullet, f, c)\in (\Delta_d\downarrow F)
			,\ (\forall c\in C,\ f: d\ to F(c))$
	\item Morphisms:\\
		$\big((\id_\bullet, h): (\bullet, f, c) \to (\bullet, f', c')\big)
			\in (\Delta_d\downarrow F),$\\
		\phantom{$\qquad$}
		$(\forall c,c'\in C,\ h:c\to c',\ f:d\to F(c),\ f':d\to F(c'))$\\
		Such that the following commutes:
		% https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMpIl0sWzEsMCwiRihjJykiXSxbMCwxLCJkIl0sWzEsMSwiZCJdLFswLDEsIkYoaCkiXSxbMiwzLCJcXG1hdGhybXtpZH1fZCIsMl0sWzAsMiwiZiIsMl0sWzEsMywiZiciXV0=
		\[\begin{tikzcd}
			{F(c)} & {F(c')} \\
			d & d
			\arrow["{F(h)}", from=1-1, to=1-2]
			\arrow["{\id_d}"', from=2-1, to=2-2]
			\arrow["f"', from=1-1, to=2-1]
			\arrow["{f'}", from=1-2, to=2-2]
		\end{tikzcd}
		=
		% https://q.uiver.app/?q=WzAsMyxbMCwwLCJGKGMpIl0sWzAsMSwiRihjJykiXSxbMSwwLCJkIl0sWzAsMSwiRihoKSIsMl0sWzEsMiwiZiciLDJdLFswLDIsImYiXV0=
		\begin{tikzcd}
			{F(c)} & d \\
			{F(c')}
			\arrow["{F(h)}"', from=1-1, to=2-1]
			\arrow["{f'}"', from=2-1, to=1-2]
			\arrow["f", from=1-1, to=1-2]
		\end{tikzcd}\]
\end{itemize}
An initial object of $(\Delta_d\downarrow F)$ is an object $(\bullet,f,c)$ such
that for any other object $(\bullet, f', c')$ there exists a unique morphism
$h:c \to c'$ such that the previous diagram commutes. This is precisely an
initial morphism from $d$ to $F$.

\subsection{Terminal morphisms}
For categories $C,D$, a terminal morphism or a universal morphism from a functor
$F: C \to D$ into an object $d\in D$ consists of an object $c\in C$ coupled with
a morphism $f: F(c) \to d$ such that for any other $c'\in C$ and $f': F(c') \to
d$, there exists a unique $h:c' \to c$ such that the following diagram commutes.

\subsubsection*{Uniqueness up to isomorphism}
A terminal morphism can be viewed as a terminal object in the comma category
$(F\downarrow \Delta_d)$, and terminal objects are unique up to isomorphism.
Therefore, terminal morphisms are unique up to isomorphism.

\subsubsection*{As the terminal object of a comma category}
Consider categories $C, D$ with a functor $F: C\to D$, together with the
discrete category $\underline{1}$ with a single element $\bullet$ and a constant
funtor $\Delta_d: \underline{1} \to D$. The comma category $(F\downarrow
\Delta_d)$ consists of:
\begin{itemize}
	\item Objects:\\
		$(c, f, \bullet)\in (F\downarrow \Delta_d)
			,\ (\forall c\in C,\ f: F(c)\to d)$
	\item Morphisms:\\
		$\big((h, \id_\bullet): (c', f', \bullet) \to (c, f, \bullet)\big)
			\in (F\downarrow \Delta_d),$\\
		\phantom{$\qquad$}
		$(\forall c,c'\in C,\ h:c'\to c,\ f:F(c)\to d,\ f':F(c')\to d)$\\
		Such that the following commutes:
		% https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMnKSJdLFsxLDAsIkYoYykiXSxbMCwxLCJkIl0sWzEsMSwiZCJdLFswLDEsIkYoaCkiXSxbMiwzLCJcXG1hdGhybXtpZH1fZCIsMl0sWzAsMiwiZiciLDJdLFsxLDMsImYiXV0=
		\[\begin{tikzcd}
			{F(c')} & {F(c)} \\
			d & d
			\arrow["{F(h)}", from=1-1, to=1-2]
			\arrow["{\id_d}"', from=2-1, to=2-2]
			\arrow["{f'}"', from=1-1, to=2-1]
			\arrow["f", from=1-2, to=2-2]
		\end{tikzcd}
		=
		% https://q.uiver.app/?q=WzAsMyxbMCwxLCJGKGMnKSJdLFswLDAsIkYoYykiXSxbMSwwLCJkIl0sWzAsMSwiRihoKSJdLFswLDIsImYnIiwyXSxbMSwyLCJmIl1d
		\begin{tikzcd}
			{F(c)} & d \\
			{F(c')}
			\arrow["{F(h)}", from=2-1, to=1-1]
			\arrow["{f'}"', from=2-1, to=1-2]
			\arrow["f", from=1-1, to=1-2]
		\end{tikzcd}\]
\end{itemize}
A terminal object of $(F\downarrow \Delta_d)$ is an object $(c, f, \bullet)$
such that for any other object $(c', f', \bullet)$ there exists a unique
morphism $h:c' \to c$ such that the previous diagram commutes. This is precisely
a terminal morphism from $F$ to $d$.
