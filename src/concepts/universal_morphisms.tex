\section{Universal Morphisms}

\subsection{Initial Morphisms}

\begin{definition}[Initial Morphism\index{Initial Morphism}]
	An initial morphism or a universal morphism from an object $d\in \D_0$ to a
	functor $F:\C\to \D$ consists of an object $c\in \C_0$ coupled with a morphism
	$f: d\to F(c)$ such that for any other $c'\in \C_0$ and $f': d \to F(c')$,
	there exists a unique $h:c\to c'$ such that the following diagram commutes:
	% https://q.uiver.app/?q=WzAsNSxbMCwwLCJjIl0sWzAsMSwiYyciXSxbMiwwLCJkIl0sWzEsMCwiRihjKSJdLFsxLDEsIkYoYycpIl0sWzAsMSwiaCIsMix7InN0eWxlIjp7ImJvZHkiOnsibmFtZSI6ImRhc2hlZCJ9fX1dLFszLDQsIkYoaCkiLDIseyJzdHlsZSI6eyJib2R5Ijp7Im5hbWUiOiJkYXNoZWQifX19XSxbMiw0LCJmJyJdLFsyLDMsImYiLDJdXQ==
	\[\begin{tikzcd}[ampersand replacement=\&]
		c \& {F(c)} \& d \\
		{c'} \& {F(c')}
		\arrow["h"', dashed, from=1-1, to=2-1]
		\arrow["{F(h)}"', dashed, from=1-2, to=2-2]
		\arrow["{f'}", from=1-3, to=2-2]
		\arrow["f"', from=1-3, to=1-2]
	\end{tikzcd}\]
\end{definition}

\begin{remark}
	Initial morphisms are the dual of terminal morphisms.
\end{remark}

\begin{theorem}[Uniqueness of Initial Morphisms\label{thm:initial_morphism_iso}]
	For an object $d\in \D_0$, and a functor $F:\C\to \D$, if there are mulpliple
	initial morphisms from $d$ to $F$ then they are all isomophic.

	\begin{proof}
		For a set-up of categories and functors:
		% https://q.uiver.app/?q=WzAsMyxbMiwwLCJcXEMiXSxbMSwwLCJcXEQiXSxbMCwwLCJcXHVuZGVybGluZXsxfSJdLFswLDEsIkYiXSxbMiwxLCJcXERlbHRhX2QiLDJdXQ==&macro_url=https%3A%2F%2Fraw.githubusercontent.com%2Faortega0703%2Fnotes-category-theory%2Fmain%2Fsrc%2Fmacros.tex
		\[\begin{tikzcd}[ampersand replacement=\&]
			{\underline{1}} \& \D \& \C
			\arrow["F", from=1-3, to=1-2]
			\arrow["{\Delta_d}"', from=1-1, to=1-2]
		\end{tikzcd}\]

		The comma category $(\Delta_d\downarrow F)$ consists of:
		\begin{itemize}
			\item Objects:
				\[\big(\forall (f:d\to F(c))\in \D_1\big)
					\big(\<0,f,c\>\in(\Delta_d\downarrow F)\big)_0\]
			\item Morphisms:
				\[
					\begin{gathered}
						\big(\forall (h: c \to c') \in \C_1
							,\ (f: d \to F(c))\in \D_1,\ (f': d\to F(c'))\in \D_1\big)\\
						\big(\<\id_0, h\>: \<0, f, c\> \to \<0, f', c'\>\big)
						\in(\Delta_d\downarrow F)_1
					\end{gathered}
				\]

				Such that the following commutes:
				% https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMpIl0sWzEsMCwiRihjJykiXSxbMCwxLCJkIl0sWzEsMSwiZCJdLFswLDEsIkYoaCkiXSxbMiwzLCJcXG1hdGhybXtpZH1fZCIsMl0sWzIsMCwiZiJdLFszLDEsImYnIiwyXV0=
				\[\begin{tikzcd}
					{F(c)} & {F(c')} \\
					d & d
					\arrow["{F(h)}", from=1-1, to=1-2]
					\arrow["{\mathrm{id}_d}"', from=2-1, to=2-2]
					\arrow["f", from=2-1, to=1-1]
					\arrow["{f'}"', from=2-2, to=1-2]
				\end{tikzcd}
				=
				% https://q.uiver.app/?q=WzAsMyxbMCwwLCJGKGMpIl0sWzAsMSwiRihjJykiXSxbMSwwLCJkIl0sWzAsMSwiRihoKSIsMl0sWzIsMSwiZiciXSxbMiwwLCJmIiwyXV0=
				\begin{tikzcd}
					{F(c)} & d \\
					{F(c')}
					\arrow["{F(h)}"', from=1-1, to=2-1]
					\arrow["{f'}", from=1-2, to=2-1]
					\arrow["f"', from=1-2, to=1-1]
				\end{tikzcd}\]
		\end{itemize}
		An initial object of $(\Delta_d\downarrow F)$ is an object $\<0,f,c\>$ such
		that for any other object $\<0, f', c'\>$ there exists a unique morphism
		$h:c \to c'$ such that the previous diagram commutes. This is precisely an
		initial morphism from $d$ to $F$. Finally, by Theorem
		\ref{thm:initial_object_iso} they are unique up to isomorphism.
	\end{proof}
\end{theorem}

\subsection{Terminal Morphisms}

\begin{definition}[Terminal Morphism\index{Terminal Morphism}]
	A terminal morphism or a universal morphism from a functor $F: \C \to \D$ into
	an object $d\in \D_0$ consists of an object $c\in \C_0$ coupled with a
	morphism $f: F(c) \to d$ such that for any other $c'\in \C_0$ and $f': F(c')
	\to d$, there exists a unique $h:c' \to c$ such that the following diagram
	commutes:
	% https://q.uiver.app/?q=WzAsNSxbMCwxLCJjIl0sWzAsMCwiYyciXSxbMiwxLCJkIl0sWzEsMSwiRihjKSJdLFsxLDAsIkYoYycpIl0sWzQsMywiRihoKSIsMix7InN0eWxlIjp7ImJvZHkiOnsibmFtZSI6ImRhc2hlZCJ9fX1dLFs0LDIsImYnIl0sWzMsMiwiZiIsMl0sWzEsMCwiaCIsMix7InN0eWxlIjp7ImJvZHkiOnsibmFtZSI6ImRhc2hlZCJ9fX1dXQ==
	\[\begin{tikzcd}[ampersand replacement=\&]
		{c'} \& {F(c')} \\
		c \& {F(c)} \& d
		\arrow["{F(h)}"', dashed, from=1-2, to=2-2]
		\arrow["{f'}", from=1-2, to=2-3]
		\arrow["f"', from=2-2, to=2-3]
		\arrow["h"', dashed, from=1-1, to=2-1]
	\end{tikzcd}\]
\end{definition}

\begin{remark}
	Terminal morphisms are the dual of initial morphisms.
\end{remark}

\begin{theorem}[Uniqueness of Terminal Morphisms\label{thm:terminal_morphism_iso}]
	For an object $d\in \D_0$, and a functor $F:C\to D$, if there are mulpliple
	initial morphisms from $d$ to $F$ then they are all isomophic.

	\begin{proof}
		For a set-up of categories and functors:
		% https://q.uiver.app/?q=WzAsMyxbMCwwLCJcXEMiXSxbMSwwLCJcXEQiXSxbMiwwLCJcXHVuZGVybGluZXsxfSJdLFswLDEsIkYiXSxbMiwxLCJcXERlbHRhX2QiLDJdXQ==&macro_url=https%3A%2F%2Fraw.githubusercontent.com%2Faortega0703%2Fnotes-category-theory%2Fmain%2Fsrc%2Fmacros.tex
		\[\begin{tikzcd}[ampersand replacement=\&]
			\C \& \D \& {\underline{1}}
			\arrow["F", from=1-1, to=1-2]
			\arrow["{\Delta_d}"', from=1-3, to=1-2]
		\end{tikzcd}\]

		The comma category $(F \downarrow \Delta_d)$ consists of:
		\begin{itemize}
			\item Objects:
				\[
					\big(\forall (f: F(c)\to d)\in \D_1\big)
					\big(\<c, f, 0\> \in (F\downarrow \Delta _d)_0\big)
				\]
			\item Morphisms:
				\[
					\begin{gathered}
						\big(\forall (h: c'\to c)\in \C_1
							,\ (f:F(c)\to d)\in \D_1,\ (f':F(c') \to d)\in\D_1\big)\\
						\big((\<h, \id_0\>: \<c', f', 0\> \to \<c, f, 0\>)\in (F\downarrow\Delta_d)_1 \big)
						\end{gathered}
				\]
				Such that the following commutes:
				% https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMnKSJdLFsxLDAsIkYoYykiXSxbMCwxLCJkIl0sWzEsMSwiZCJdLFswLDEsIkYoaCkiXSxbMiwzLCJcXG1hdGhybXtpZH1fZCIsMl0sWzAsMiwiZiIsMl0sWzEsMywiZiciXV0=
				\[\begin{tikzcd}
					{F(c')} & {F(c)} \\
					d & d
					\arrow["{F(h)}", from=1-1, to=1-2]
					\arrow["{\mathrm{id}_d}"', from=2-1, to=2-2]
					\arrow["f"', from=1-1, to=2-1]
					\arrow["{f'}", from=1-2, to=2-2]
				\end{tikzcd}
				=
				% https://q.uiver.app/?q=WzAsMyxbMCwwLCJGKGMpIl0sWzAsMSwiRihjJykiXSxbMSwwLCJkIl0sWzEsMCwiRihoKSJdLFsxLDIsImYnIiwyXSxbMCwyLCJmIl1d
				\begin{tikzcd}
					{F(c)} & d \\
					{F(c')}
					\arrow["{F(h)}", from=2-1, to=1-1]
					\arrow["{f'}"', from=2-1, to=1-2]
					\arrow["f", from=1-1, to=1-2]
				\end{tikzcd}\]
		\end{itemize}
		A terminal object of $(F\downarrow \Delta_d)$ is an object $\<c,f,0\>$ such
		that for any other object $(c', f', 0)$ there exists a unique morphism $h:c'
		\to c$ such that the previous diagram commutes. This is precisely a terminal
		morphism from $F$ to $d$. Finally, by Theorem \ref{thm:terminal_object_iso}
		they are unique up to isomorphism.
	\end{proof}
\end{theorem}

\begin{definition}[Universal Morphism\index{Universal Morphism}]
	A universal morphism is either an initial or terminal morphism.
	\parencite{lane:working_mathematician}
\end{definition}
