\section{Universal Morphisms}

\begin{definition}[Initial Morphism]
	For categories $C,D$, an initial morphism or a universal morphism from an
	object $d\in D$ to a functor $F:C\to D$ consists of an object $c\in C$ coupled
	with a morphism $f: d\to F(c)$ such that for any other $c'\in C$ and $f': d
	\to F(c')$, there exists a unique $h:c\to c'$ such that the following diagram
	commutes:

		c & {F(c)} & d \\
		{c'} & {F(c')}
		\arrow[""{name=0, anchor=center, inner sep=0}, "h"', dashed, from=1-1, to=2-1]

\begin{theorem}[Uniqueness up to Isomorphism of Initial Morphisms]
	For categories $C,D$, an object $d\in D$, and a functor $F:C\to D$, if there
	are mulpliple initial morphisms from $d$ to $F$ then they are all isomophic.

	\begin{proof}
		For a configuration of categories and functors:
		% https://q.uiver.app/?q=WzAsMyxbMiwwLCJDIl0sWzEsMCwiRCJdLFswLDAsIlxcdW5kZXJsaW5lezF9Il0sWzAsMSwiRiJdLFsyLDEsIlxcRGVsdGFfZCIsMl1d
		\[\begin{tikzcd}
			{\underline{1}} & D & C
			\arrow["F", from=1-3, to=1-2]
			\arrow["{\Delta_d}"', from=1-1, to=1-2]
		\end{tikzcd}\]

		The comma category $(\Delta_d\downarrow F)$ consists of:
		\begin{itemize}
			\item Objects:\\
				$(0, f, c)\in (\Delta_d\downarrow F) ,\ (\forall c\in C,\ f: d\to
					F(c))$
			\item Morphisms:\\
				$(\forall c,c'\in C,\ h:c\to c',\ f:d\to F(c),\ f':d\to F(c'))$\\
				\phantom{$\qquad$}
				$\big((\id_0, h): (0, f, c) \to (0, f', c')\big) \in
					(\Delta_d\downarrow F),$\\
				Such that the following commutes:
				% https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMpIl0sWzEsMCwiRihjJykiXSxbMCwxLCJkIl0sWzEsMSwiZCJdLFswLDEsIkYoaCkiXSxbMiwzLCJcXG1hdGhybXtpZH1fZCIsMl0sWzIsMCwiZiJdLFszLDEsImYnIiwyXV0=
				\[\begin{tikzcd}
					{F(c)} & {F(c')} \\
					d & d
					\arrow["{F(h)}", from=1-1, to=1-2]
					\arrow["{\mathrm{id}_d}"', from=2-1, to=2-2]
					\arrow["f", from=2-1, to=1-1]
					\arrow["{f'}"', from=2-2, to=1-2]
				\end{tikzcd}
				=
				% https://q.uiver.app/?q=WzAsMyxbMCwwLCJGKGMpIl0sWzAsMSwiRihjJykiXSxbMSwwLCJkIl0sWzAsMSwiRihoKSIsMl0sWzIsMSwiZiciXSxbMiwwLCJmIiwyXV0=
				\begin{tikzcd}
					{F(c)} & d \\
					{F(c')}
					\arrow["{F(h)}"', from=1-1, to=2-1]
					\arrow["{f'}", from=1-2, to=2-1]
					\arrow["f"', from=1-2, to=1-1]
				\end{tikzcd}\]
		\end{itemize}
		An initial object of $(\Delta_d\downarrow F)$ is an object $(0,f,c)$ such
		that for any other object $(0, f', c')$ there exists a unique morphism $h:c
		\to c'$ such that the previous diagram commutes. This is precisely an
		initial morphism from $d$ to $F$. Finally, by Theorem
		\ref{thm:init_obj_iso} they are unique up to isomorphism.
	\end{proof}
\end{theorem}

\begin{definition}[Terminal Morphism]
	For categories $C,D$, a terminal morphism or a universal morphism from a
	functor $F: C \to D$ into an object $d\in D$ consists of an object $c\in C$
	coupled with a morphism $f: F(c) \to d$ such that for any other $c'\in C$ and
	$f': F(c') \to d$, there exists a unique $h:c' \to c$ such that the following
	diagram commutes:

		{c'} & {F(c')}
		\arrow[""{name=0, anchor=center, inner sep=0}, "{F(h)}", dashed, from=2-2, to=1-2]

\begin{theorem}[Uniqueness up to Isomorphism of Terminal Morphisms]
	For categories $C,D$, an object $d\in D$, and a functor $F:C\to D$, if there
	are mulpliple initial morphisms from $d$ to $F$ then they are all isomophic.

	\begin{proof}
		For a configuration of categories and functors:
		% https://q.uiver.app/?q=WzAsMyxbMCwwLCJDIl0sWzEsMCwiRCJdLFsyLDAsIlxcdW5kZXJsaW5lezF9Il0sWzAsMSwiRiJdLFsyLDEsIlxcRGVsdGFfZCIsMl1d
		\[\begin{tikzcd}
			C & D & {\underline{1}}
			\arrow["F", from=1-1, to=1-2]
			\arrow["{\Delta_d}"', from=1-3, to=1-2]
		\end{tikzcd}\]

		The comma category $(F \downarrow \Delta_d)$ consists of:
		\begin{itemize}
			\item Objects:\\
				$(c, f, 0)\in (F \downarrow \Delta_d) ,\ (\forall c\in C,\ f: F(c)\to
				d)$
			\item Morphisms:\\
				$(\forall c,c'\in C,\ h:c'\to c,\ f:F(c)\to d,\ f':F(c')\to d)$\\
				\phantom{$\qquad$}
				$\big((h, \id_0): (c', f', 0) \to (c, f, 0)\big)$\\
				Such that the following commutes:
				% https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGMnKSJdLFsxLDAsIkYoYykiXSxbMCwxLCJkIl0sWzEsMSwiZCJdLFswLDEsIkYoaCkiXSxbMiwzLCJcXG1hdGhybXtpZH1fZCIsMl0sWzAsMiwiZiIsMl0sWzEsMywiZiciXV0=
				\[\begin{tikzcd}
					{F(c')} & {F(c)} \\
					d & d
					\arrow["{F(h)}", from=1-1, to=1-2]
					\arrow["{\mathrm{id}_d}"', from=2-1, to=2-2]
					\arrow["f"', from=1-1, to=2-1]
					\arrow["{f'}", from=1-2, to=2-2]
				\end{tikzcd}
				=
				% https://q.uiver.app/?q=WzAsMyxbMCwwLCJGKGMpIl0sWzAsMSwiRihjJykiXSxbMSwwLCJkIl0sWzEsMCwiRihoKSJdLFsxLDIsImYnIiwyXSxbMCwyLCJmIl1d
				\begin{tikzcd}
					{F(c)} & d \\
					{F(c')}
					\arrow["{F(h)}", from=2-1, to=1-1]
					\arrow["{f'}"', from=2-1, to=1-2]
					\arrow["f", from=1-1, to=1-2]
				\end{tikzcd}\]
		\end{itemize}
		A terminal object of $(F\downarrow \Delta_d)$ is an object $(c,f,0)$ such
		that for any other object $(c', f', 0)$ there exists a unique morphism $h:c'
		\to c$ such that the previous diagram commutes. This is precisely a terminal
		morphism from $F$ to $d$. Finally, by Theorem \ref{thm:term_obj_iso} they
		are unique up to isomorphism.
	\end{proof}
\end{theorem}

\begin{definition}[Universal Morphism]
	A universal morphism is either an initial or terminal morphism.
	\parencite{maclane:working_mathematician}
\end{definition}
