\section{Functors}

\subsection{Definition}
A functor can be thought of as an structure preserving transformation between
categories, or a way to identify one category inside another. A functor $F: C
\to D$ from a category $C$ to a category $D$ consist of a mapping of:
\parencite{fong_spivak:7sketches}

\begin{itemize}
  \item Object to objects\\
    $F(c) \in D,\ (\forall c \in C)$
  \item Morphism to morphisms\\
    $(F(f): F(c) \to F(c')) \in D,
      \ (\forall c,c' \in C,\ f : c \to c')$
\end{itemize}

Such that it preserves:
\begin{itemize}
  \item Identity\\
    $F(\mathrm{id}_c) = \mathrm{id}_{F(c)},
      \ (\forall c \in C)$
  \item Composition\\
    $F(g \circ f) = F(g) \circ F(f),
      \ (\forall a,b,c \in C,\ f: a \to b,\ g: b \to c)$
\end{itemize}

\subsection{Functor Properties}

\subsubsection*{Faithful}
For categories $C, D$, a functor $F:C\to D$ is said to be faithful when for
every $x,y\in C$, the function $F_{x, y}: \mathrm{hom}_C(x, y) \to
\mathrm{hom}_D(F(x), F(y))$ is injective. \parencite{awodey:category_theory}

\subsubsection*{Full}
For categories $C, D$, a functor $F:C\to D$ is said to be full when for every
$x,y\in C$, the function $F_{x, y}: \mathrm{hom}_C(x, y) \to
\mathrm{hom}_D(F(x), F(y))$ is surjective.
\parencite{adamek_herrlich_strecker:joy_cats}

\subsubsection*{Endofunctor}
For a category $C$, a functor $F:C\to C$ (one from $C$ to itself) is said to be
an endofunctor. \parencite{adamek_herrlich_strecker:joy_cats}

\subsubsection*{Bifunctor}
For categories $A, B, C$, a functor $F:A\times B \to C$ (one with $2$
parameters) is said to be a bifunctor.
\parencite{maclane:working_mathematician}

\subsection{As morphisms between categories}
For functors to be considered as morphisms between categories they must satisty:

\subsubsection*{Identity}
For a category $C$ there exists an identity functor $\mathrm{id}_C: C \to C$
that consists of maps of:
\begin{itemize}
  \item Objects:\\
    $\mathrm{id}_C(c) = c \in C,\ (\forall c\in C)$
  \item Morphisms:\\
    $\mathrm{id}_C(f): c\to c' = f \in C,\ (\forall c, c'\in C,\ f:c\to c')$
\end{itemize}

\subsubsection*{Composition}
For categories $A,B,C$ and functors $F:A\to B$ and $G:B\to C$ the composite
functor $G\circ F$ consists of maps of:
\begin{itemize}
  \item Objects:\\
    $(G\circ F)(a) = G(F(a)) \in C,\ (\forall a\in A)$
  \item Morphisms:\\
    $(G\circ F)(f): (G\circ F)(a) \to (G\circ F)(a') = G(F(f)) \in C,\ (\forall
      a, a'\in A,\ f:a\to a')$
\end{itemize}

\subsubsection*{Unitality and Associativity}
For categories $C,D$ a functor $F:C\to D$ consists of two different functions,
one from $\mathrm{ob}_C$ to $\mathrm{ob}_D$ and one from $\mathrm{hom}_C$ to
$\mathrm{hom}_D$. As function composition fulfills unitality and associativity
then so does functor composition
