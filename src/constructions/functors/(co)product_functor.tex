\subsection{Product and Coproduct Functors}

\subsubsection*{Product Functor}
  \begin{definition}[Product Functor\index{Product Functor}]
    For a category $\C$ with all binary products, it is possible to define a
    functor $\times:\C\times \C \to \C$ composed by maps
    of~\parencite[p.~47]{awodey:category_theory}:
    \begin{itemize}
      \item Objects:
        \[\big(\forall (c,c')\in (\C\times \C)_0\big)
          \big(\times(c, c') = c\times c'\big)\]
      \item Morphisms:
        \[\big(\forall (f, f')\in (C\times C)_1\big)
          \big(\times(f, f') = f\times f'\big)\]
    \end{itemize}
  \end{definition}

  \begin{theorem}[Left Adjoint of Product]
    For a category $\C$, the left adjoint of the product functor $\times:
    \C\times \C\to \C$ is the diagonal functor $\Delta:\C\to \C\times \C$.

    \begin{proof}
      See proof of Theorem \ref{thm:diagonal_right}
    \end{proof}
  \end{theorem}

\subsubsection*{Coproduct Functor}
  \begin{definition}[Coproduct Functor]
    For a category $\C$ with all binary coproducts, it is possible to define a
    functor $+:\C\times \C \to \C$ composed by maps
    of~\parencite[p.~61]{awodey:category_theory}:
    \begin{itemize}
      \item Objects:
        \[\big(\forall (c,c')\in (C\times C)_0\big)
          \big(+(c, c') = c+ c'\big)\]
      \item Morphisms:
        \[\big(\forall (f, f')\in (C\times C)_1\big)
          \big(+(f, f') = f+ f'\big)\]
    \end{itemize}
  \end{definition}

  \begin{theorem}[Right Adjoint of $+$]
    For a category $C$, the right adjoint of the coproduct functor $+: \C\times
    \C\to \C$ is the diagonal functor $\Delta:\C\to \C\times \C$.

    \begin{proof}
      See proof of Theorem \ref{thm:diagonal_left}
    \end{proof}
  \end{theorem}
