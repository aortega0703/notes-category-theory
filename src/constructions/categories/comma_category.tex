\subsection{Comma Category}
\begin{definition}[Comma Category\index{Comma Category}]
  For functors $F:\A\to \C, G:\B\to \C$, the comma category $(F\downarrow G)$
  consists of:~\parencite{adamek_herrlich_strecker:joy_cats}
  \begin{itemize}
    \item Objects:
      \[\big(\forall a\in \A_0,\ b\in \B_0,\ (f: F(a) \to G(b))\in \C_1\big)
        \big(\< a, f, b\>\in (F\downarrow G)\big)\]
    \item Morphisms:
      \[
        \begin{gathered}
          \big(\forall f, f'\in \C_1,
            \ (g: a \to a')\in \A_1,\ (h: b\to b')\in \B_1\big)\\
          \big((\<g, h\> : \<a, f, b\> \to \<a', f', b'\>)\in \C_1\big)
        \end{gathered}
      \]
      Such that the following diagram commutes:
      % https://q.uiver.app/?q=WzAsNCxbMCwwLCJGKGEpIl0sWzEsMCwiRihhJykiXSxbMCwxLCJHKGIpIl0sWzEsMSwiRyhiJykiXSxbMCwxLCJGKGcpIl0sWzIsMywiRyhoKSIsMl0sWzAsMiwiZiIsMl0sWzEsMywiZiciXV0=
      \[\begin{tikzcd} {F(a)} & {F(a')} \\
        {G(b)} & {G(b')}
        \arrow["{F(g)}", from=1-1, to=1-2]
        \arrow["{G(h)}"', from=2-1, to=2-2]
        \arrow["f"', from=1-1, to=2-1]
        \arrow["{f'}", from=1-2, to=2-2]
      \end{tikzcd}\]
  \end{itemize}
\end{definition}
