\subsection{Cone and Cocone Categories}
\begin{definition}[Cone Category\index{Cone Category}]
  For a functor $F:C\to D$, the cone category $\mathrm{Cone}(F)$ consists of:
  \parencite{fong_spivak:7sketches}

  \begin{itemize}
    \item Objects:\\
      $\big(d, \big(d_c: d \to F(c)\big)_{c\in C}\big)$.
    \item Morphisms:\\
      $f: (d, d_*) \to (d', d'_*)$, for all $(f:d\to d')\in D$ such
      that for every $c\in C$ the following commutes:
      % https://q.uiver.app/?q=WzAsMyxbMCwwLCJkIl0sWzEsMCwiZCciXSxbMSwxLCJGKGMpIl0sWzAsMiwiZF9jIiwyXSxbMSwyLCJkJ19jIl0sWzAsMSwiZiJdXQ==
      \[\begin{tikzcd}
        d & {d'} \\
        & {F(c)}
        \arrow["{d_c}"', from=1-1, to=2-2]
        \arrow["{d'_c}", from=1-2, to=2-2]
        \arrow["f", from=1-1, to=1-2]
      \end{tikzcd}\]
  \end{itemize}
\end{definition}

\begin{definition}[Cocone Category\index{Cocone Category}]
  For a functor $F:C\to D$, the cocone category $\mathrm{Cocone}(F)$ consists
  of:

  \begin{itemize}
    \item Objects:\\
      $\big(d, \big(d_c: F(c) \to d\big)_{c\in C}\big)$.
    \item Morphisms:\\
      $f: (d, d_*) \to (d', d'_*)$, for all $(f:d\to d')\in D$ such
      that for every $c\in C$ the following commutes:
      % https://q.uiver.app/?q=WzAsMyxbMCwxLCJkIl0sWzEsMSwiZCciXSxbMCwwLCJGKGMpIl0sWzIsMCwiZF9jIiwyXSxbMiwxLCJkJ19jIl0sWzAsMSwiZiIsMl1d
      \[\begin{tikzcd}[ampersand replacement=\&]
        {F(c)} \\
        d \& {d'}
        \arrow["{d_c}"', from=1-1, to=2-1]
        \arrow["{d'_c}", from=1-1, to=2-2]
        \arrow["f"', from=2-1, to=2-2]
      \end{tikzcd}\]
  \end{itemize}
\end{definition}
