\subsection{Natural Number Objects}
\begin{definition}[Natural Number Object\index{Natural Number Object}]
	For a terminal object $1$, a natural number object candidate is an object $n$
	with morphisms $q: 1 \to a$ and $p: a \to a$:
	% https://q.uiver.app/?q=WzAsMyxbMCwwLCIxIl0sWzEsMCwiYSJdLFsyLDAsImEiXSxbMCwxLCJxIl0sWzEsMiwicCJdXQ==
	\[\begin{tikzcd}[ampersand replacement=\&]
		1 \& a \& a
		\arrow["q", from=1-1, to=1-2]
		\arrow["p", from=1-2, to=1-3]
	\end{tikzcd}\]

	A natural number object is a candidate ($n$, $z$, $s$) such that for any
	cantidate ($a$, $q$, $p$) there exists a unique $h$ that makes the following
	commute~\parencite{awodey:category_theory}:
	% https://q.uiver.app/?q=WzAsNSxbMCwwLCIxIl0sWzEsMCwibiJdLFsxLDEsImEiXSxbMiwwLCJuIl0sWzIsMSwiYSJdLFswLDEsInoiXSxbMCwyLCJxIiwyXSxbMSwzLCJzIl0sWzIsNCwicCIsMl0sWzEsMiwiaCIsMCx7InN0eWxlIjp7ImJvZHkiOnsibmFtZSI6ImRhc2hlZCJ9fX1dLFszLDQsImgiLDAseyJzdHlsZSI6eyJib2R5Ijp7Im5hbWUiOiJkYXNoZWQifX19XV0=
	\[\begin{tikzcd}[ampersand replacement=\&]
		1 \& n \& n \\
		\& a \& a
		\arrow["z", from=1-1, to=1-2]
		\arrow["q"', from=1-1, to=2-2]
		\arrow["s", from=1-2, to=1-3]
		\arrow["p"', from=2-2, to=2-3]
		\arrow["h", dashed, from=1-2, to=2-2]
		\arrow["h", dashed, from=1-3, to=2-3]
	\end{tikzcd}\]
\end{definition}
